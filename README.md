Доорх Matrix гэсэн классыг гүйцээж загварчлах.



class Matrix

{

 private:

 int _m, _n;

 float **_values;

 public:

 Matrix(int m=1,int n=1);

 Matrix(const Matrix &matrix);

 ~Matrix();

 int getRow() const;

 int getColumn() const;

void removeColumn(int i); //i-р мөр устгах

void removeRow(int j); //j -р баганыг устгах

 void setValues(**values);

 float** getValues() const;

 void setValue(int i, int j, float value);//i-р мөр j-р баганын утгад value утгыг оноох

 float getValue(int i, int j) const; //i-р мөр j-р баганын утгыг буцаах

 void resize(int n, int m);// хүснэгтийн хэмжээсийг дахин шинэчлэх, хүснэгтэд дахин санах ой нөөцлөөд хүснэгтийн хуучинтайгаа давхардаж буй дугаартай бол хуучин утгууд үгүй бол тэг утга оноох

};

2. * үржих (хоёр матрицийг үржээд үржвэрийг буцаана)

3. - хасах (хоёр матрицийг хасаад ялгаварыг буцаана)

6. = нэг матрицийн объектыг нөгөөд утга оноох

7. ++ нэгээр нэмэгдүүлэх

8. -- нэгээр хорогдуулах

9. += 

10. -= 

11. *=
